# vuex的使用例子

# vuex 填写 dashboardStore.js(单个)

import dashboardService from "@/services/dashboardService"; // 封装的接口
var dashboardStore = {
    namespaced: true, //使用命名空间，这样只在局部使用
    state: {
        searchContent: '', // 首页-搜索内容
        allSearchResultData: [] // 首页的展示内容
    },
    mutations: {
        //首页-搜索参数 赋值
        changeSearchContent: (state, newVal) => {
            state.searchContent = newVal;
        },
        //首页-搜索参数 清空
        removeSearchContent: (state) => {
            state.searchContent = ''
        },
    },
    actions: {
        // 通过接口返回 赋值
        textSearchForPage: async ({ commit, state }, payLoad) => {
            let params = {
                text: state.searchContent,
                desc: payLoad.desc,
            };
            const { data, code } = await dashboardService.resultDosearch(params);
            state.allSearchResultData = data.rows;
        }
    },
};

export default dashboardStore;

# 监听修改搜索参数，同时修改vuex上的储存参数

import { mapState } from "vuex";

export default {
    data() {
        return {
            obj:''
        }
    },
    computed: {
        ...mapState("dashboardStore", {
            searchJump: (state) => state.searchJump,
        }),
    },
    watch: {
        searchJump: {
            handler(newval, oldval) {
                let obj = this.obj // 搜索参数
                // 事件处理函数 修改vuex内的数据
                this.$store.commit("dashboardStore/changeSearchContent", obj);
            },
            deep: true,
            immediate: true,
        },
    },
}

# 调用vuex上的查询接口

import {  mapActions } from "vuex";
export default {
    data() {
        return {
            searchContent:''
        }
    },
    methods: {
        // 格式化日期
        ...mapActions("dashboardStore", ["textSearchForPage"]),
        search(){
            this.$store.commit(
                "dashboardStore/changeSearchContent",
                this.searchContent
            );
            let params = {
                searchContent: this.searchContent,
                desc: false,
            };
            this.textSearchForPage(params)
        }
    }
}

# 刷新的保存数据

在vue的app.vue的created函数添加代码：

if (sessionStorage.getItem("pteTagsNav")) {
    this.$store.replaceState(
        Object(
            {},
            this.$store.state.pte_tagsNav.homeName,
            sessionStorage.getItem("pteTagsNav")
        )
    );
}
window.addEventListener("beforeunload", () => {
    sessionStorage.setItem(
        "pteTagsNav",
        this.$store.state.pte_tagsNav.homeName
    );
});
