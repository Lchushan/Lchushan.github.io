# 按钮权限

​ 与后端配合，给设置权限的时候，通过后端的接口储存-开通了的按钮权限缩代表的字符串，在登录的时候，通过接口获取当前用户的按钮权限列表

​ 前端封装自定义指令，来进行项目整体的按钮的判断展示（当前按钮代表的字符串是否在接口返回的用户拥有的权限列表中）

​ 代码如下：

把接口返回的字段进行前端重新命名，防止之后修改的方便 permissionAllData.js

const PermissionAllData = {
    BtnAdd: "Btn_add",  // 前端重新命名
    ...
};
export { PermissionAllData }

全局的权限校验方法 main.js

Vue.prototype.$checkAuthority = (permissionCode) => {
    let hasPermission = false;  // 初始化权限为 false
    const permissionLocalStorage = JSON.parse(localStorage.getItem('permissionData')) || [] // 从本地储存中获取权限数组
    if (permissionLocalStorage instanceof Array && permissionLocalStorage.length > 0) {
        permissionLocalStorage.forEach(item => {
            if (item === permissionCode) {
                hasPermission = true;
            }
        })
    }
    return hasPermission;
}

进行对比 authorityall.js

import { PermissionAllData } from './permissionAllData'
import Vue from 'pte-ui/main' // vue插件

export default {
    inserted(el, binding, vnode) {
        const { value } = binding;
        const val = PermissionAllData[value]  //改造后打开

        const hasPermission = Vue.prototype.$checkAuthority(val); // 进行对比

        if (!hasPermission) {
            if (el.parentNode) {
                el.parentNode.removeChild(el);
            } else {
                el.innerHTML = "";
            }
        } else {
            el && el.setAttribute("code", value);
        }
    },
};

自定义指令 authorityIndex.js

import authority from './authorityall';

const install = function(Vue) {
    Vue.directive('authority', authority);
};

if (window.Vue) {
    window.authority = authority;
    Vue.use(install);
}

authority.install = install;
export default authority;

挂在在vue上 main

import AuthorityDirective from './authorityIndex.js';  // 自定义指令
Vue.use(AuthorityDirective);

使用

<el-button v-authority="'BtnAdd'" type="primary"}}</el-button>
