            .main {

            }

    <script>
        // var arr1 = ["何润东", "刘德华", "刘诗诗", "刘涛", "张涵予", "张杰", "张若昀", "张雨绮", "周星驰", "周雨彭", "胡歌", "刘亦菲", "娜扎", "杨幂", "张含韵", "张靓颖", "赵丽颖", "周冬雨", "熊希霖", "范冰冰"];
        var arr1 = ["炒面", "螺蛳粉", "蒸菜", "香锅", "兰州拉面", "烧腊", "外婆饭"];
        var allNameP = document.getElementById("allName");
        // allNameP.innerHTML = arr1.toString();
        generateMenu()
        // 生成菜单
        function generateMenu() {
            var HTML = ''
            for (let index = 0; index < arr1.length; index++) {
                const element = arr1[index];
                HTML += `<div>${index+1}. ${element}</div>`
            }
            allNameP.innerHTML = HTML
        }

        // 数组长度为 arr1.length
        function bianhuan() {
            var ran1 = Math.round(Math.random() * 100);  //生成50以内的随机数 
            ran1 += arr1.length; // 至少循环一轮
            console.log(ran1);
            var num = 0;
            var inte = setInterval(function() {
                var img1 = document.getElementsByTagName("img")[0];
                var p1 = document.getElementById("name");
                var imgName = arr1[num % arr1.length];
                img1.src = "./imgs/" + imgName + ".jpeg";
                p1.innerHTML = imgName;
                if (num == ran1) {
                    clearInterval(inte);
                }
                num++;
            }, 100);
        }
    </script>
