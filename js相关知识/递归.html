# 递归事件

调用事件

// 对象递归事件
findIdByName(tree) {
    var name = []
    var clssData = {};
    function find(obj) {
        if (obj != null) {
            if (obj.classId) {
                name.unshift(obj.name)
                clssData[obj.classId] = obj.instId || obj.id;
            }
            find(obj.parentId);
        }
    }
    find(tree);
    return { clssData, name };
},

  5. 递归开始：生成未知层数的评论，parent 组件内部递归

    1. 递归组件语法, 必须起一个 name 然后就能在自己的 template 当中调用自己

    2. 自己调用自己的数据传递模式是一样的 也是 props

    3. v-if="parentData.parent" 就是递归终止处理或计算的出口

      <template>
  <div>
    {{parentData.content}}
 
//组件自己调用自己，有parent就调用渲染并传递数据，没有就不渲染
 
    <parent v-if="parentData.parent" :parentData="parentData.parent">{{parentData.conent}}</parent>
  </div>
</template>
 
<script>
export default {
//递归组件语法，name
  name: "parent",
  props: ["parentData"]
