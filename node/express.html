# express

# express的介绍：

基于node.js平台，快速、开放、极简的Web开发框架。

express的框架核心特性：

  1. 可以设置中间件来响应HTTP请求。
  2. 定义了路由表用于执行不同的HTTP请求动作。
  3. 可以通过向模板传递参数来动态渲染HTML页面。

# express的使用：

安装：

npm install express -S;     //先安装express框架

使用：

// 1. 引入框架模块
var express = require('express')

// 2. 创建服务器
var app = express();

// 3. 开启服务器并监听端口
app.listen(3000,()=>{
    console.log('express server is running at http://127.0.0.1:3000');
})

// 4.给app对象注册中间件,监听请求
// use是一个中间件,可以用来监听get和post请求
app.use((req,res)=>{
    // res.end('hello world') // 返回英文数据
    // res.writeHeader(200,{
    //   'Content-Type':'text/plan;charset=utf-8'
    // })
    // res.end('express框架的基本使用...');// 直接返回带有中文的内容会有乱码 加响应头后可以解决

    res.send('express框架的基本使用...') ;// 可以直接返回中文并解析不会出现乱码
})

# 接收get和post的不同的请求

使用：

// 1. 引入框架模块
var express = require('express')

// 2. 创建服务器
var app = express();

// 3. 开启服务器并监听端口
app.listen(3000,()=>{
    console.log('express server is running at http://127.0.0.1:3000');
})

// 4.给app对象注册中间件,监听请求
// use是一个中间件,可以用来监听get和post请求
app.get('/',(req,res)=>{
    res.send('这是一个index的主页面...')
})

app.get('/movie', (req, res) => {
    res.send('这是一个电影的主页面...')
})

app.post('/login', (req, res) => {
    res.send('这是一个登陆的主页面...')
})

# EJS模板引擎介绍

express框架中推荐使用的页面模板引擎是ejs，而不是art-template

使用步骤：

  1. 使用命令 npm i ejs -S 安装这个模板引擎
  2. 给express框架设置默认的模板引擎
  3. 设置模板引擎文件的默认路径

使用：

// 1. 引入模块
var express =require('express')

// 2. 创建服务器对象
var app = express()
// 通过app.set方法，设置express的默认模板引擎为ejs
// 第一个参数是'view engine' 是固定的写法,表示我们要设置默认的模板引擎
app.set('view engine','ejs')
// 通过app.set('views',path)来设置默认模板页面的存放路径;
// 第一个参数'views'是固定写法
// 注意,在express中,已经默认约定好了,如果用户不设置app.set('views',path), 
// 那么,默认的模板 文件存放路径就是'./views',就是当前根目录下面的views文件夹,约定大于配置
app.set('views','./views')

// 3. 启动服务器并监听端口
app.listen(3000,()=>{
    console.log('express server is running at http://127.0.0.1:3000');
})

// 4. 给app对象注册中间件,并监听请求
app.get('/',(req,res)=>{
    // 注意:ejs模板引擎,默认的模板文件名后缀名不是.html,而是ejs
    res.render('index.ejs',{})
})

app.get('/movie', (req, res) => {
    res.render('movie.ejs', {})
})

# express.static托管静态页面

利用Express内置的express.static可以方便地托管静态文件,例如图片、CSS、JavaScript文件等。

将静态资源 文件所在的目录作为参数传递给express.static中间件就可以提供静态资源文件的访问了.

例如,假设在public目录放置了图片、CSS和JavaScript文件,你就可以:

app.use(express.static('public'));

现在public目录下面的文件就可以访问了。

http://localhost:3000/images/kitten.jpg
http://localhost:3000/css/style.css
http://localhost:3000/js/app.js
http://localhost:3000/images/bg.png
http://localhost:3000/images/hello.html
//所有文件的路径都是相对于存储目录的,因此存放静态文件的目录名不会出现在URL中.

如果你希望所有通过express.static访问的文件都存放在一个"虚拟(virtual)"目录(即目录根本不存在)下面,可以通过为静态资源目录指定一个挂载路径的方式来实现,如下所示:

app.use('/static'express.static('public'));

现在，你就可以通过带有“/static"前缀的地址来访问public目录下面的文件了.

http://localhost:3000/static/images/kitten.jpg
http://localhost:3000/static/css/style.css
http://localhost:3000/static/js/app.js
http://localhost:3000/static/images/bg.png
http://localhost:3000/static/images/hello.html

# 使用include引入公共部分

使用ejs模板引擎中的include()方法来引入即可. <%- %>表示不转义输出.

<%- include('./layout/header.ejs')%>
    <h2>这是一个主页面的内容</h2>
    <p>姓名:<%= name%></p>
    <p>性别:<%= gender%></p>
    <p>年龄:<%= age%></p>
    <p>爱好:</p>
    <ul>
        <% hobby.forEach(item=>{ %>
            <li><%= item%></li>
        <%})%>
    </ul>
    <img src="/a/b/public/images/psb.jpg" alt="">
<%- include('./layout/footer.ejs')%>

# express.Router创建路由

app.route()

可以使用app.router()创建路由路径的链式路由句柄.由于路径在一个地方指定,这样做有助于创建模块化的路由.

let express = require('express')

let router = express.Router();

router.get('/',(req,res)=>{
    res.render('index',{})
})
    .get('/add',(req,res)=>{
    res.render('add',{})
})
    .get('/info',(req,res)=>{
    res.render('info',{})
})
    .get('/edit',(req,res)=>{
    res.render('edit',{})
})
    .post('/detail',(req,res)=>{
    res.send('这是post请求发送的数据')
})
module.exports = router;

express.Router

可使用express.Router类创建模块化、可挂载的路由句柄。Router实例是一个完整的中间件和路由系统，因此常称其为一个"mini-app"。

在项目目录下创建名为router.js的文件，内容如下:

// 1. 导入模块
var express = require('express')

// 2. 创建一个路由模块
var router = express.Router();

// 将请求路径挂载到router路由对象上
router.get('/', (req, res) => {
    // res.send('这是主页面的内容...')
    res.render('index', { name: 'tom', age: 20, gender: '男', hobby: ['吃饭', '睡觉', '打豆豆'] })
})
router.get('/movie', (req, res) => {
    // res.send('这是一个电影页面的内容啊...')
    res.render('movie', {})
})
router.get('/login', (req, res) => {
    // res.send('这是一个用户登陆的请求啊......')
    res.render('login', {})
})

router.post('/', (req, res) => {
    res.send('这是一个post请求啊...')
})
// 将创建的router暴露出去
module.exports = router

在app.js中的内容如下:

// 1. 引入模块
var express = require('express')
var ejs = require('ejs')
var router = require('./route.js')

// 2. 创建服务器对象
var app = express();

// 3. 启动服务,并监听端口
app.listen(3000,()=>{
    console.log('express server is running at http://127.0.0.1:3000');
})

app.set('view engine','ejs');
app.set('views','./views')
ejs.delimiter = '?';
// 实现静态资源的托管 
// app.use(express.static('public'))
app.use('/a/b/public',express.static('public')) ;// 可以放在任何虚拟目录里面,而不会受到影响
app.use('/c/d/assets',express.static('assets'))

// 4. 注册路由模块
app.use(router)
