            this.bloodSessionId = response.data.blood.sessionId;

              ...response.data.blood.relations

              ...response.data.blood.nodes

            let newArr = [];
            let obj = {};
            for (var i = 0; i < arr.length; i++) {
              if (!obj[arr[i].instId]) {
                newArr.push(arr[i]);
                obj[arr[i].instId] = true;
